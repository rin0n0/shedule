            <span class="subject-text" :class="{ 'red-text': lesson.status === 'cancellation' }">
                {{ lesson.subject }}
            </span>
            <div class="indicators">
                <span v-if="hasNote" class="note-indicator" title="–ï—Å—Ç—å –∑–∞–º–µ—Ç–∫–∞">üìù</span>
                <span v-if="lesson.is_stream" class="dot blue" title="–ü–æ—Ç–æ–∫"></span>
                <span v-if="lesson.status === 'cancellation'" class="dot red" title="–û—Ç–º–µ–Ω–∞"></span>
                <span v-else-if="lesson.status === 'replacement'" class="dot yellow" title="–ó–∞–º–µ–Ω–∞"></span>
            </div>
        </div>

        <!-- –ù–∏–∂–Ω–∏–π –±–ª–æ–∫: –ü—Ä–µ–ø–æ–¥–∞–≤–∞—Ç–µ–ª—å/–≥—Ä—É–ø–ø–∞ –∏ —à–µ–≤—Ä–æ–Ω -->
        <div class="card-footer">
            <span class="sub-text">
                {{ !mainStore.userTeacher ? lesson.teacher : lesson.group }}
            </span>
            <div class="footer-right">
                <span v-if="lesson.subgroup !== 0 && mainStore.isRegistredSubgroup" class="subgroup-label">

            </div>
        </div>

    </div>
</template>

<script setup lang="ts">
import { computed } from 'vue';
import type { Lesson } from '@/types';
import { useMainStore } from '@/stores/main_store';

const props = defineProps<{
    lesson: Lesson,
    date: string
}>();
const emit = defineEmits(['lessonClick']);
const mainStore = useMainStore();

const hasNote = computed(() => {
    const key = `${props.date}_${props.lesson.lesson_number}_${props.lesson.subject}`;
    return !!mainStore.notes[key];
});

const handleClick = () => {
    emit('lessonClick', { lesson: props.lesson, date: props.date });
};
</script>
